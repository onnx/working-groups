module OPAdd
  use Tensor
  use int.Int
  use real.Real
  use std.Cfloat

  let ghost function dadd (a b : data real) : data real
    = fun ks -> a ks + b ks


  let ghost function opadd (a b : tensor real) : tensor real
    requires { a ~= b }
    ensures { result ~= a }
    ensures { result.data = dadd a.data b.data }
    = { dims = a.dims ;
        data = dadd a.data b.data ;
        background = a.background + b.background }

end
